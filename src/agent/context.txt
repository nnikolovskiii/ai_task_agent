"""
Voice Handler for AI Task Agent

This module provides the actual voice processing capabilities including:
- Speech-to-text (STT) using OpenAI Whisper
- Text-to-speech (TTS) using OpenAI TTS
- Voice command processing and validation
- Audio format handling
"""

import os
import asyncio
import logging
import tempfile
from typing import Dict, Any, Optional, Tuple
import aiohttp
import json
from datetime import datetime

# Try to import OpenAI
try:
    import openai
    OPENAI_AVAILABLE = True
except ImportError:
    OPENAI_AVAILABLE = False
    logging.warning("OpenAI not available. Voice processing will be disabled.")

# Try to import audio processing libraries
try:
    import numpy as np
    import soundfile as sf
    AUDIO_AVAILABLE = True
except ImportError:
    AUDIO_AVAILABLE = False
    logging.warning("Audio processing libraries not available.")


class VoiceProcessor:
    """
    Voice processing handler that provides STT and TTS capabilities.
    """
    
    def __init__(self, api_key: str = None):
        self.logger = logging.getLogger(__name__)
        self.api_key = api_key or os.getenv("OPENAI_API_KEY")
        
        if not self.api_key:
            raise ValueError("OpenAI API key required for voice processing")
            
        if not OPENAI_AVAILABLE:
            raise ImportError("OpenAI package not available")
            
        openai.api_key = self.api_key
        
        self.logger.info("Voice processor initialized")
    
    async def process(self, audio_input: bytes) -> Dict[str, Any]:
        """
        Process audio input and return transcribed text.
        
        Args:
            audio_input: Raw audio data as bytes
            
        Returns:
            Dict with transcription results and metadata
        """
        try:
            # Convert raw audio to proper format if needed
            audio_data = await self._prepare_audio(audio_input)
            
            # Use OpenAI Whisper for transcription
            transcript = await self._transcribe_audio(audio_data)
            
            return {
                "success": True,
                "text": transcript,
                "confidence": 0.95,  # OpenAI doesn't provide confidence scores
                "metadata": {
                    "model": "whisper-1",
                    "timestamp": datetime.now().isoformat(),
                    "audio_size": len(audio_input)
                }
            }
            
        except Exception as e:
            self.logger.error(f"Voice processing failed: {e}")
            return {
                "success": False,
                "error": str(e),
                "text": "",
                "metadata": {}
            }
    
    async def _prepare_audio(self, audio_input: bytes) -> bytes:
        """
        Prepare audio data for processing.
        
        Args:
            audio_input: Raw audio bytes
            
        Returns:
            Processed audio bytes in proper format
        """
        # For now, assume the input is already in a supported format
        # In a real implementation, you might need to:
        # 1. Convert to proper format (WAV, 16kHz, 16-bit, mono)
        # 2. Normalize audio levels
        # 3. Remove silence
        
        return audio_input
    
    async def _transcribe_audio(self, audio_data: bytes) -> str:
        """
        Transcribe audio using OpenAI Whisper.
        
        Args:
            audio_data: Audio bytes
            
        Returns:
            Transcribed text
        """
        # Create temporary file for audio
        with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as temp_file:
            temp_path = temp_file.name
            temp_file.write(audio_data)
        
        try:
            # Use OpenAI's Whisper API
            with open(temp_path, "rb") as audio_file:
                response = openai.Audio.transcribe(
                    model="whisper-1",
                    file=audio_file,
                    response_format="text"
                )
            
            # Clean up temp file
            os.unlink(temp_path)
            
            return response.strip()
            
        except Exception as e:
            # Clean up temp file
            if os.path.exists(temp_path):
                os.unlink(temp_path)
            raise e
    
    async def text_to_speech(self, text: str, voice: str = "alloy") -> bytes:
        """
        Convert text to speech using OpenAI TTS.
        
        Args:
            text: Text to convert
            voice: Voice to use (alloy, echo, fable, onyx, nova, shimmer)
            
        Returns:
            Audio bytes
        """
        try:
            response = openai.Audio.speech.create(
                model="tts-1",
                voice=voice,
                input=text
            )
            
            # Get audio bytes
            audio_data = response.content
            
            return audio_data
            
        except Exception as e:
            self.logger.error(f"Text-to-speech failed: {e}")
            raise
    
    async def process_command(self, command_text: str, 
                          context: Dict[str, Any] = None) -> Dict[str, Any]:
        """
        Process a text command and determine the appropriate action.
        
        Args:
            command_text: The transcribed command
            context: Additional context for processing
            
        Returns:
            Dict with command analysis and response
        """
        command_text = command_text.lower().strip()
        
        # Simple command parsing
        if "create task" in command_text or "new task" in command_text:
            # Extract task details
            task_details = self._extract_task_details(command_text)
            return {
                "action": "create_task",
                "parameters": task_details,
                "response_text": f"I'll create a new task: {task_details.get('description', 'No description provided')}"
            }
        
        elif "list tasks" in command_text or "show tasks" in command_text:
            return {
                "action": "list_tasks",
                "parameters": {},
                "response_text": "I'll show you your current tasks."
            }
        
        elif "complete task" in command_text or "finish task" in command_text:
            # Extract task ID or description
            task_ref = self._extract_task_reference(command_text)
            return {
                "action": "complete_task",
                "parameters": {"task_ref": task_ref},
                "response_text": f"I'll mark task {task_ref} as complete."
            }
        
        else:
            return {
                "action": "unknown",
                "parameters": {"query": command_text},
                "response_text": f"I heard: {command_text}. Could you please clarify what you'd like me to do?"
            }
    
    def _extract_task_details(self, command_text: str) -> Dict[str, str]:
        """Extract task details from voice command."""
        # Simple extraction - in real implementation, use NLP
        parts = command_text.split("create task")[-1].split("new task")[-1]
        
        # Try to extract priority
        priority = "medium"
        if "high priority" in parts:
            priority = "high"
        elif "low priority" in parts:
            priority = "low"
        
        # Clean up description
        description = parts.strip()
        description = description.replace("high priority", "").replace("low priority", "").strip()
        
        return {
            "description": description,
            "priority": priority
        }
    
    def _extract_task_reference(self, command_text: str) -> str:
        """Extract task reference from command."""
        # Simple extraction - look for numbers or keywords
        import re
        numbers = re.findall(r'\d+', command_text)
        if numbers:
            return numbers[0]
        
        # Return the remaining text as task description
        return command_text.replace("complete task", "").replace("finish task", "").strip()
    
    async def health_check(self) -> Dict[str, Any]:
        """
        Check if voice processing is working correctly.
        
        Returns:
            Dict with health status
        """
        try:
            # Test TTS
            test_text = "Hello, this is a health check."
            audio = await self.text_to_speech(test_text)
            
            # Test STT

                "dependencies": {
                    "openai": OPENAI_AVAILABLE,
                    "audio": AUDIO_AVAILABLE
                }
            }
            
        except Exception as e:
            return {
                "healthy": False,
                "error": str(e),
                "tts_working": False,
                "stt_working": False
            }
