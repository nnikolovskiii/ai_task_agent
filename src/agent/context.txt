from __future__ import annotations

import os

from dotenv import load_dotenv
from langgraph.constants import START, END
from langgraph.graph import StateGraph

from google import genai
from google.genai import types
import httpx

from src.agent import State

load_dotenv()
api_key = os.getenv("GEMINI_API_KEY")


def process_pdf_with_gemini(state: State):
    """
    Retrieves a PDF from a URL and processes it with Google's Gemini model.
    """
    doc_url = state["file_url"]

    # Initialize the Google Gemini client
    client = genai.Client(api_key)

    # Retrieve and encode the PDF bytes
    doc_data = httpx.get(doc_url).content

    prompt = "Summarize this document"
    response = client.models.generate_content(
        model="gemini-2.5-flash",
        contents=[
            types.Part.from_bytes(
                data=doc_data,
                mime_type='application/pdf',
            ),
            prompt
        ]
    )

    print(response.text)
    return {}

# Create the graph
optimizer_builder = StateGraph(State)

# Add the PDF processing node
optimizer_builder.add_node("process_pdf_with_gemini", process_pdf_with_gemini)

# Connect the node to the graph
optimizer_builder.add_edge(START, "process_pdf_with_gemini")
optimizer_builder.add_edge("process_pdf_with_gemini", END)

graph = optimizer_builder.compile()
